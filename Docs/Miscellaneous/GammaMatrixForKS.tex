\end{split}

\begin{equation}
\begin{split}

\end{split}
\end{equation}
where
\textcolor[rgb]{0,0,1}{

\begin{split}

\textcolor[rgb]{0,0,1}{

\end{split}
\end{equation}

\begin{equation}
\begin{split}
